-- Executar uma consulta que mostre todas as colunas da tabela instagram quando a carreira for diferente de jogador(a) de futebol (Footballer)
SELECT * FROM instagram WHERE career != 'Footballer'; 

-- Executar uma consulta que mostre todas as colunas da tabela instagram quando os seguidores em milhões for maior que 200 milhões
SELECT * FROM instagram WHERE followersMillions > 200; 

-- Executar uma consulta que mostre todas as colunas da tabela instagram quando os seguidores em milhões for menor que 155 milhões
SELECT * FROM instagram WHERE followersMillions < 155; 

-- Executar uma consulta que mostre todas as colunas da tabela instagram quando os seguidores em milhões for maior ou igual a 200 milhões
SELECT * FROM instagram WHERE followersMillions >= 200; 

-- Executar uma consulta que mostre as colunas primeiro nome, sobrenome, referência no instagram, seguidores em milhões da tabela instagram quando os seguidores em milhões for menor ou igual a 155 milhões
SELECT firstName, surname, instagramHandle, followersMillions FROM instagram WHERE followersMillions <= 155; 

-- Executar uma consulta que mostre todas as colunas da tabela instagram quando a carreira for cantor(a) (singer) e os seguidores em milhões for maior que 200 milhões
SELECT * 
FROM instagram 
WHERE career = 'Singer' 
      AND followersMillions > 200; 

-- Executar uma consulta que mostre todas as colunas da tabela instagram quando a carreira for jogador(a) de futebol (Footballer) ou os seguidores em milhões for menor que 150 milhões
SELECT * 
FROM instagram 
WHERE career = 'Footballer' 
      OR followersMillions < 150;
