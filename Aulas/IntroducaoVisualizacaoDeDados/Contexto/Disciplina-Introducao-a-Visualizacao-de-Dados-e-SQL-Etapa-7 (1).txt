Comandos para analisar os dados da tabela STATION_DATA e será usada a cláusula ORDER BY. 
SELECT count(*)
FROM station_data;         -- 28.000 registros 

SELECT * 
FROM station_data; 

SELECT station_number
FROM station_data; 

SELECT station_number
FROM station_data
ORDER BY station_number; 

SELECT distinct station_number
FROM station_data;         -- 6.368 valores diferentes

SELECT distinct station_number
FROM station_data
ORDER BY station_number;      

SELECT distinct station_number
FROM station_data
ORDER BY station_number DESC;

SELECT distinct report_code
FROM station_data
ORDER BY report_code;      -- 27.973 valores diferentes

SELECT distinct year
FROM station_data
ORDER BY year;            -- 79 valores diferentes (primeiro ano: 1930 e último ano: 2010)

SELECT distinct month
FROM station_data
ORDER BY month;        -- 12 valores diferentes (entre 1 e 12) 

SELECT distinct day
FROM station_data
ORDER BY day;             -- 31 valores diferentes (entre 1 e 31)

SELECT distinct dew_point
FROM station_data
ORDER BY dew_point;  -- 1076 valores diferentes (o primeiro: -49.9 e o último: 85.8)

SELECT min(dew_point), max(dew_point)
FROM station_data; 

SELECT distinct station_pressure
FROM station_data
ORDER BY station_pressure;     -- 1374 valores diferentes (o primeiro: 602.3 e o último: 1040.5)

SELECT min(station_pressure), max(station_pressure)
FROM station_data; 

SELECT distinct visibility
FROM station_data
ORDER BY visibility;        -- 431 valores diferentes (o primeiro: 0 e o último: 99.4)

SELECT distinct wind_speed
FROM station_data
ORDER BY wind_speed;     -- 302 valores diferentes (o primeiro: 0 e o último: 56.7)

SELECT distinct temperature
FROM station_data
ORDER BY temperature;        -- 1174 valores diferentes (o primeiro: -72.9 e o último: 106.8)

SELECT distinct precipitation
FROM station_data
ORDER BY precipitation;    -- 149 valores diferentes (o primeiro: 0 e o último: 7.48)

SELECT distinct snow_depth
FROM station_data
ORDER BY snow_depth;    -- 94 valores diferentes (o primeiro: 0.4 e o último: 108.3)

SELECT distinct fog
FROM station_data
ORDER BY fog;        -- 2 (0, 1)

SELECT distinct rain
FROM station_data
ORDER BY rain;        -- 2 (0, 1)

SELECT distinct hail
FROM station_data
ORDER BY hail;        -- 2 (0, 1)

SELECT distinct thunder
FROM station_data
ORDER BY  thunder;    -- 2 (0, 1)

SELECT distinct tornado
FROM station_data
ORDER BY  tornado;    -- 2 (0, 1)

SELECT count(*) 
FROM station_data
WHERE station_number is NULL;         -- 0 

SELECT count(*) 
FROM station_data
WHERE report_code is NULL;       -- 0 

SELECT count(*) 
FROM station_data
WHERE month = 2 
     AND day = 30;                -- 0 
     
SELECT distinct day
FROM station_data
WHERE month = 2 
ORDER BY day;         -- 29 valores diferentes (o primeiro: 1 e o último: 29)
